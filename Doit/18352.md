#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int N{}, M{}, K{}, X{};
	cin >> N >> M >> K >> X;

	vector<int> distances(N+1);
	vector<vector<int>> graph(N+1);

	int num1{}, num2{};
	for (int i{}; i < M; i++)
	{
		cin >> num1 >> num2;
		graph[num1].push_back(num2);
	}

	queue<int> bfs;
	vector<bool> isVisit(N + 1);
	isVisit[X] = true;
	bfs.push({X});
	while (!bfs.empty())
	{
		int node{ bfs.front()};
		bfs.pop();

		for (auto& e : graph[node])
		{
			if (!isVisit[e])
			{
				bfs.push({ e });
				isVisit[e] = true;
				distances[e] = distances[node] + 1;
			}
		}
	}

	bool isExist{};
	for (int i{ 1 }; i <= N; i++)
	{
		if (distances[i] == K)
		{
			cout << i << '\n';
			isExist = true;
		}
	}

	if (!isExist)
	{
		cout << -1;
	}
}
